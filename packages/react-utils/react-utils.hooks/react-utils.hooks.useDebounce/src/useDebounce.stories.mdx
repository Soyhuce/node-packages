import { useState } from 'react'
import { Meta, Canvas, Story } from '@storybook/addon-docs'

import Readme from '../README.md'
import { useDebounce } from './useDebounce.ts'

<Meta title='react-utils/Hooks/useDebounce' />

<Readme />

export const Template = (args) => {
  const [value, setValue] = useState('')
  const setValueDebounced = useDebounce(setValue, args.delay)
  return (
    <div>
      <input onChange={(e) => setValueDebounced(e.target.value)} placeholder='Please type here' />
      <div>{value}</div>
    </div>
  )
}

<Canvas>
  <Story name='Playground' args={{ delay: 500 }}>
    {Template.bind({})}
  </Story>
</Canvas>
